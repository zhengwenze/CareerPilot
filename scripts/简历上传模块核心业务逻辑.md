```python
# validator.py
from pathlib import Path


class ResumeValidationError(Exception):
    pass


class ResumeFileValidator:
    ALLOWED_SUFFIXES = {".pdf"}
    MAX_FILE_SIZE = 10 * 1024 * 1024  # 10MB

    @classmethod
    def validate(cls, file_name: str, file_bytes: bytes) -> None:
        suffix = Path(file_name).suffix.lower()
        if suffix not in cls.ALLOWED_SUFFIXES:
            raise ResumeValidationError("Only PDF files are supported.")

        if not file_bytes:
            raise ResumeValidationError("Uploaded file is empty.")

        if len(file_bytes) > cls.MAX_FILE_SIZE:
            raise ResumeValidationError("File size exceeds 10MB limit.")
```

这对应清单里的“只支持 PDF、做格式和大小校验”。

## 2）PDF 文本抽取

```python
# extractor.py
from io import BytesIO
from pypdf import PdfReader


class ResumeExtractionError(Exception):
    pass


class PdfTextExtractor:
    @staticmethod
    def extract_text(file_bytes: bytes) -> str:
        try:
            reader = PdfReader(BytesIO(file_bytes))
            texts: list[str] = []

            for page in reader.pages:
                page_text = page.extract_text() or ""
                if page_text.strip():
                    texts.append(page_text)

            full_text = "\n".join(texts).strip()
            if not full_text:
                raise ResumeExtractionError("No readable text found in PDF.")

            return full_text
        except Exception as e:
            raise ResumeExtractionError(f"Failed to extract PDF text: {e}") from e
```

注意：

* 这只适合**文本型 PDF**
* 扫描版图片 PDF 基本抽不出来
* 第一版可以接受，后面再接 OCR

这也符合清单“先规则化、可解释，先别上 AI 增强”。

## 3）文本清洗

PDF 抽出来的文本很脏，不清洗规则会很难写。

```python
# cleaner.py
import re


class ResumeTextCleaner:
    @staticmethod
    def clean(text: str) -> str:
        text = text.replace("\u00a0", " ")
        text = text.replace("\t", " ")
        text = re.sub(r"[ ]{2,}", " ", text)
        text = re.sub(r"\n{3,}", "\n\n", text)
        text = re.sub(r"[•●▪■◆]", "-", text)
        return text.strip()

    @staticmethod
    def to_lines(text: str) -> list[str]:
        lines = [line.strip() for line in text.splitlines()]
        return [line for line in lines if line]
```

## 5）section 切分器

这里很关键。不要一上来就写复杂 NLP，先按“标题行切块”。

```python
# splitter.py
from collections import defaultdict
from .patterns import SECTION_ALIASES


class ResumeSectionSplitter:
    @staticmethod
    def normalize_heading(text: str) -> str:
        return text.strip().lower().replace("：", "").replace(":", "")

    @classmethod
    def identify_section(cls, line: str) -> str | None:
        normalized = cls.normalize_heading(line)
        for section_name, aliases in SECTION_ALIASES.items():
            alias_set = {a.lower() for a in aliases}
            if normalized in alias_set:
                return section_name
        return None

    @classmethod
    def split(cls, lines: list[str]) -> dict[str, list[str]]:
        sections: dict[str, list[str]] = defaultdict(list)
        current_section = "header"

        for line in lines:
            detected = cls.identify_section(line)
            if detected:
                current_section = detected
                continue

            sections[current_section].append(line)

        return dict(sections)
```

输出大概这样：

```python
    "header": [...],
    "education": [...],
    "work_experience": [...],
    "projects": [...],
```

## 6）规则解析器

这是最核心业务代码。

```python
# parser.py
import re
from copy import deepcopy

from .patterns import CITY_KEYWORDS, TECH_KEYWORDS, TOOL_KEYWORDS, LANGUAGE_KEYWORDS
from .schemas import EMPTY_RESUME_DATA


EMAIL_RE = re.compile(r"[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}")
PHONE_RE = re.compile(r"(?:\+?\d[\d\s-]{7,}\d)")


class ResumeRuleParser:
    @staticmethod
    def dedupe_keep_order(items: list[str]) -> list[str]:
        seen: set[str] = set()
        result: list[str] = []

        for item in items:
            normalized = re.sub(r"\s+", " ", item.strip()).lower()
            if not normalized or normalized in seen:
                continue
            seen.add(normalized)
            result.append(item.strip())

        return result

    @staticmethod
    def extract_email(text: str) -> str:
        match = EMAIL_RE.search(text)
        return match.group(0) if match else ""

    @staticmethod
    def extract_phone(text: str) -> str:
        match = PHONE_RE.search(text)
        return match.group(0).strip() if match else ""

    @staticmethod
    def extract_location(text: str) -> str:
        lower_text = text.lower()
        for city in CITY_KEYWORDS:
            if city.lower() in lower_text:
                return city
        return ""

    @staticmethod
    def extract_name(header_lines: list[str]) -> str:
        for line in header_lines[:5]:
            stripped = line.strip()
            if not stripped:
                continue
            if EMAIL_RE.search(stripped) or PHONE_RE.search(stripped):
                continue
            if len(stripped) > 32:
                continue
            return stripped
        return ""

    @staticmethod
    def extract_summary(header_lines: list[str]) -> str:
        candidates: list[str] = []
        for line in header_lines[1:8]:
            stripped = line.strip()
            if not stripped:
                continue
            if EMAIL_RE.search(stripped) or PHONE_RE.search(stripped):
                continue
            candidates.append(stripped)
            if len(candidates) >= 3:
                break
        return " ".join(candidates)

    @staticmethod
    def extract_section_lines(section_lines: list[str]) -> list[str]:
        return [line.strip("- ").strip() for line in section_lines if line.strip()]

    @classmethod
    def extract_skills(cls, skill_lines: list[str]) -> dict[str, list[str]]:
        joined = "\n".join(skill_lines)

        technical = [kw for kw in TECH_KEYWORDS if kw.lower() in joined.lower()]
        tools = [kw for kw in TOOL_KEYWORDS if kw.lower() in joined.lower()]
        languages = [kw for kw in LANGUAGE_KEYWORDS if kw.lower() in joined.lower()]

        return {
            "technical": cls.dedupe_keep_order(technical),
            "tools": cls.dedupe_keep_order(tools),
            "languages": cls.dedupe_keep_order(languages),
        }

    @classmethod
    def parse(cls, raw_text: str, sections: dict[str, list[str]]) -> dict:
        data = deepcopy(EMPTY_RESUME_DATA)

        header_lines = sections.get("header", [])
        full_text = raw_text

        data["basic_info"]["name"] = cls.extract_name(header_lines)
        data["basic_info"]["email"] = cls.extract_email(full_text)
        data["basic_info"]["phone"] = cls.extract_phone(full_text)
        data["basic_info"]["location"] = cls.extract_location(full_text)
        data["basic_info"]["summary"] = cls.extract_summary(header_lines)

        data["education"] = cls.dedupe_keep_order(
            cls.extract_section_lines(sections.get("education", []))
        )
        data["work_experience"] = cls.dedupe_keep_order(
            cls.extract_section_lines(sections.get("work_experience", []))
        )
        data["projects"] = cls.dedupe_keep_order(
            cls.extract_section_lines(sections.get("projects", []))
        )
        data["skills"] = cls.extract_skills(sections.get("skills", []))
        data["certifications"] = cls.dedupe_keep_order(
            cls.extract_section_lines(sections.get("certifications", []))
        )

        return data
```

这段代码就是把清单中这些规则编程化了：

* 姓名：前 5 行里第一个合理短文本
* 邮箱：邮箱正则
* 电话：电话正则
* 地点：城市关键词
* 摘要：前几行非联系方式内容
* 教育/工作/项目/技能/证书：按 section heading 切块抽取
* 技能分类：关键词分类
* 去重：大小写归一化 + 空格清理。

## 7）总流程编排

```python
# pipeline.py
from dataclasses import dataclass

from .validator import ResumeFileValidator
from .extractor import PdfTextExtractor
from .cleaner import ResumeTextCleaner
from .splitter import ResumeSectionSplitter
from .parser import ResumeRuleParser


@dataclass
class ResumeParseResult:
    raw_text: str
    cleaned_text: str
    sections: dict[str, list[str]]
    structured_json: dict


class ResumeParsingPipeline:
    @classmethod
    def run(cls, file_name: str, file_bytes: bytes) -> ResumeParseResult:
        ResumeFileValidator.validate(file_name=file_name, file_bytes=file_bytes)

        raw_text = PdfTextExtractor.extract_text(file_bytes)
        cleaned_text = ResumeTextCleaner.clean(raw_text)
        lines = ResumeTextCleaner.to_lines(cleaned_text)
        sections = ResumeSectionSplitter.split(lines)
        structured_json = ResumeRuleParser.parse(cleaned_text, sections)

        return ResumeParseResult(
            raw_text=raw_text,
            cleaned_text=cleaned_text,
            sections=sections,
            structured_json=structured_json,
        )
```

# 八、这套代码里真正最重要的不是抽文本，而是“可解释规则”

这份清单其实已经把方向定死了：**第一版不要搞黑盒 AI，先把规则引擎做出来。**

所以你代码里要坚持这几个原则：

## 1. section 先切，再抽字段

不要一大坨全文本上来就乱匹配。
正确顺序是：

* 先抽全文
* 再按标题切 section
* 再在 section 内抽字段

这样误判率低很多。

## 2. 原文保留优先

像教育、工作、项目，第一版就别拆太细。
清单里建议直接用字符串数组保留原文关键信息，这很对。

因为你现在最重要的是：

* 能提取出来
* 用户能看懂
* 后续能人工修正

不是一次性全自动拆成极细粒度字段。

## 3. 技能分类用白名单

不要做复杂实体识别，先词表匹配：

* technical
* tools
* languages

这是第一版成本最低、最稳的方案。

## 4. 一定保留 `raw_text`

后续人工校正、问题排查、规则迭代，全靠它。

---

# 九、你现在最容易踩的坑

## 1. 扫描版 PDF 抽不到字

`pypdf` 只适合文本型 PDF。
很多用户的简历是导出 PDF，没问题；但如果是拍照扫描版，就会失败。

第一版建议策略：

* 抽不到文本就直接返回 `"No readable text found in PDF"`
* 前端提示用户上传可复制文本的 PDF

## 2. section 标题不统一

有人写：

* 教育背景
* 教育经历
* Education
* Academic Background

所以 alias 词典要持续补。

## 3. 姓名误识别

有些简历第一行可能是：

* 求职简历
* RESUME
* Frontend Engineer

所以 `extract_name()` 后面最好再补一些排除词，例如：

```python
INVALID_NAME_CANDIDATES = {
    "简历", "个人简历", "resume", "curriculum vitae", "frontend engineer"
}
```

## 4. 工作/项目 section 会混排

第一版不用追求 100% 精确。
先把 section 文本整段保留成数组项，后续再增强。

# 十、建议你下一步马上补的两个增强点

## 增强 1：把 section 内容进一步合并成条目

现在 section 里每一行都单独成一条，可能太碎。
你可以按“时间行”或“公司/学校行”做聚合。

例如：

* 遇到包含 `2021-2024`、`2021.06 - 2024.03` 的行，就作为一条经历起点
* 后续若干 bullet 合并到这一条里

这会让 `education` 和 `work_experience` 更像人能读的结果。

## 增强 2：给每个字段加 `confidence`

例如：

```python
{
  "basic_info": {
    "name": {"value": "张三", "confidence": 0.85},
    "email": {"value": "a@b.com", "confidence": 0.99}
  }
}
```

第一版不一定要做，但这对后续前端“人工校正体验”很有帮助。
